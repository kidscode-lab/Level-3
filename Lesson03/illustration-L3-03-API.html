JSON Data Processing in Python

üìù How Python Processes the JSON Response:

import requests import json # Get the JSON response response = requests.get("https://api.adviceslip.com/advice") data = response.json() # Display the full JSON structure print("=== Full JSON Response ===") print(json.dumps(data, indent=2)) # Iterate through the JSON data print("\n=== Processing JSON Data ===") for key, value in data.items(): print(f"Key: '{key}' ‚Üí Value: {value}") # Check if value is a nested object/array if isinstance(value, dict): print(f" ‚îî‚îÄ‚îÄ '{key}' is a nested object:") for sub_key, sub_value in value.items(): print(f" {sub_key}: {sub_value}") elif isinstance(value, list): print(f" ‚îî‚îÄ‚îÄ '{key}' is a list with {len(value)} items:") for i, item in enumerate(value): print(f" [{i}]: {item}") # Extract specific values (example) print("\n=== Extracting Specific Data ===") if 'slip' in data: advice = data['slip']['advice'] print(f"Advice: {advice}")
