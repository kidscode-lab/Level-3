ğŸš— Think of a Car Blueprint

class Pet:
    def __init__(self, name):
        self.name = name
        self.hunger = 5
        self.happiness = 5
        self.energy = 5
        self.health = 10
        self.cleanliness = 5
    
    def _keep_in_bounds(self, value, min_val=0, max_val=10):
        # Helper method to keep values in range
        return max(min_val, min(max_val, value))
    
    def get_mood(self):
        # Calculate overall mood based on all stats
        total_wellness = (
            self.health + 
            self.happiness + 
            self.energy + 
            (10 - self.hunger) + 
            self.cleanliness
        ) / 5
        
        if total_wellness >= 8:
            return "ğŸŒŸ Excellent"
        elif total_wellness >= 6:
            return "ğŸ˜Š Happy"
        elif total_wellness >= 4:
            return "ğŸ˜ Okay"
        elif total_wellness >= 2:
            return "ğŸ˜Ÿ Worried"
        else:
            return "ğŸ˜¢ Very Sad"
    
    def feed(self):
        if self.hunger > 0:
            self.hunger -= 2
            self.happiness += 1
            self.health += 1
            self.cleanliness -= 1  # Eating can be messy!
            
            # Keep all values in bounds
            self.hunger = self._keep_in_bounds(self.hunger)
            self.happiness = self._keep_in_bounds(self.happiness)
            self.health = self._keep_in_bounds(self.health)
            self.cleanliness = self._keep_in_bounds(self.cleanliness)
            
            print(f"ğŸ½ï¸ {self.name} enjoys a delicious meal!")
        else:
            print(f"ğŸ˜‹ {self.name} is already full!")
    
    def play(self):
        if self.energy > 1:
            self.energy -= 2
            self.happiness += 3
            self.hunger += 1
            self.cleanliness -= 2  # Playing gets dirty!
            
            # Keep all values in bounds
            self.energy = self._keep_in_bounds(self.energy)
            self.happiness = self._keep_in_bounds(self.happiness)
            self.hunger = self._keep_in_bounds(self.hunger)
            self.cleanliness = self._keep_in_bounds(self.cleanliness)
            
            print(f"ğŸ¾ {self.name} has a blast playing!")

        self.cleanliness = self._keep_in_bounds(self.cleanliness)
